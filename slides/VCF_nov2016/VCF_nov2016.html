$$\begin{align} \frac{\partial\rho}{\partial t} + \nabla \cdot (\rho \mathbf{u}) &= 0\\ \rho \frac{\partial \mathbf{u}}{\partial t} + \rho (\mathbf{u} \cdot \nabla)\mathbf{u} &= \nabla \cdot [-p \mathbf{I} + \mathbf{\tau}] + \mathbf{F}\\ \rho C_p \left(\frac{\partial T}{\partial t} + (\mathbf{u} \cdot \nabla) T \right) &= - (\nabla \cdot \mathbf{q}) + \mathbf{\tau} : \mathbf{S} - \frac{T}{\rho} \frac{\partial \rho}{\partial T}\bigg|_p \left(\frac{\partial p}{\partial t} + (\mathbf{u} \cdot \nabla) p\right) + \mathbf{Q} \end{align}$$
